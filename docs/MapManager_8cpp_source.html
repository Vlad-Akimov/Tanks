27bool MapManager::loadMaps() {
28 maps.clear();
29
30 try {
31 std::filesystem::create_directories(mapsDirectory);
32
33 if (!std::filesystem::exists(mapsDirectory)) {
34 std::cerr << "Directory does not exist: " << mapsDirectory << std::endl;
35 createDefaultMaps();
36 return false;
37 }
38
39 // Ищем все .map файлы в директории
40 for (const auto& entry : std::filesystem::directory_iterator(mapsDirectory)) {
41 if (entry.is_regular_file() && entry.path().extension() == ".map") {
42 std::cout << "Found map file: " << entry.path() << std::endl;
43
44 MapInfo mapInfo;
45 mapInfo.filename = entry.path().filename().string();
46
47 // Читаем информацию о карте из файла
48 std::ifstream file(entry.path());
49 if (file.is_open()) {
50 std::string line;
51
52 // Читаем displayName (первая строка)
53 if (std::getline(file, line)) {
54 mapInfo.displayName = line;
55 std::cout << "DisplayName: " << mapInfo.displayName << std::endl;
56 } else {
57 std::cerr << "Error: map file is empty or corrupted: " << mapInfo.filename << std::endl;
58 continue;
59 }
60
61 // Читаем description (вторая строка)
62 if (std::getline(file, line)) {
63 mapInfo.description = line;
64 std::cout << "Description: " << mapInfo.description << std::endl;
65 }
66
67 // Читаем размеры (третья строка)
68 if (std::getline(file, line)) {
69 std::istringstream iss(line);
70 if (!(iss >> mapInfo.width >> mapInfo.height)) {
71 std::cerr << "Error reading map dimensions from file: " << mapInfo.filename << std::endl;
72 continue;
73 }
74 std::cout << "Size: " << mapInfo.width << "x" << mapInfo.height << std::endl;
75 }
76
77 // Читаем layout (остальные строки)
78 mapInfo.layout.clear();
79 int linesRead = 0;
80 while (std::getline(file, line)) {
81 // Пропускаем пустые строки
82 if (line.empty()) continue;
83
84 if (line.length() == static_cast<size_t>(mapInfo.width)) {
85 std::vector<char> row(line.begin(), line.end());
86 mapInfo.layout.push_back(row);
87 linesRead++;
88 } else if (!line.empty()) {
89 std::cout << "Skipped line with incorrect length: " << line.length() << " instead of " << mapInfo.width << " in file " << mapInfo.filename << std::endl;
90 }
91 }
92
93 file.close();
94
95 // Проверяем корректность данных
96 if (mapInfo.layout.size() == static_cast<size_t>(mapInfo.height)) {
97 maps.push_back(mapInfo);
98 std::cout << "Successfully loaded map: " << mapInfo.displayName << " (" << linesRead << " lines)" << std::endl;
99 } else {
100 std::cerr << "Incorrect number of lines in map " << mapInfo.filename << ": " << mapInfo.layout.size() << " instead of " << mapInfo.height << std::endl;
101 }
102 } else {
103 std::cerr << "Failed to open file: " << entry.path() << std::endl;
104 }
105 }
106 }
107
108 // Если карт нет, создаем несколько примеров
109 if (maps.empty()) {
110 std::cout << "No maps found, creating default maps..." << std::endl;
111 createDefaultMaps();
112 } else {
113 std::cout << "Total maps loaded: " << maps.size() << std::endl;
114 }
115
116 return true;
117
118 } catch (const std::exception& e) {
119 std::cerr << "Error loading maps: " << e.what() << std::endl;
120 createDefaultMaps();
121 return false;
122 }

331void MapManager::createWorldFromMap(GameWorld& world, const MapInfo& map) {
332 int level = world.getCurrentLevel();
333
334 // СОХРАНЯЕМ состояние игрока
335 PlayerTank* oldPlayer = world.getPlayer();
336 int savedScore = oldPlayer ? oldPlayer->getScore() : 0;
337 int savedLives = oldPlayer ? oldPlayer->getLives() : 3;
338 int savedHealth = oldPlayer ? oldPlayer->getHealth() : 3;
339
340 // Получаем доступ к объектам мира для модификации
341 auto& objects = const_cast<std::vector<std::unique_ptr<GameObject>>&>(world.getObjects());
342 auto& projectiles = const_cast<std::vector<std::unique_ptr<Projectile>>&>(world.getProjectiles());
343 auto& bonuses = const_cast<std::vector<std::unique_ptr<Bonus>>&>(world.getBonuses());
344
345 // Очищаем ВСЕ объекты КРОМЕ игрока
346 projectiles.clear();
347 bonuses.clear();
348
349 // Удаляем все объекты кроме игрока
350 objects.erase(std::remove_if(objects.begin(), objects.end(),
351 [oldPlayer](const std::unique_ptr<GameObject>& obj) {
352 return obj.get() != oldPlayer;
353 }), objects.end());
354
355 // ВОССТАНАВЛИВАЕМ состояние игрока
356 if (oldPlayer) {
357 oldPlayer->setPosition(Point(map.width / 2, map.height - 3));
358 oldPlayer->setScore(savedScore);
359 oldPlayer->setLives(savedLives);
360 oldPlayer->setHealth(savedHealth);
361 }
362
363 int baseEnemyCount = countEnemiesInLayout(map.layout);
364 int adjustedEnemyCount = calculateAdjustedEnemyCount(baseEnemyCount, level);
365 int enemyDifficulty = calculateEnemyDifficulty(level);
366
367 int enemyCount = 0;
368 int obstacleCount = 0;
369 std::vector<Point> allValidPositions;
370 std::vector<Point> originalEnemyPositions;
371
372 for (int y = 0; y < map.height; y++) {
373 for (int x = 0; x < map.width; x++) {
374 char cell = map.layout[y][x];
375 Point pos(x, y);
376
377 if (cell == 'E') {
378 originalEnemyPositions.push_back(pos);
379 }
380
381 if (isValidEnemyPosition(pos, map, oldPlayer)) {
382 allValidPositions.push_back(pos);
383 }
384 }
385 }
386
387 std::vector<Point> enemyPositions;
388
389 for (const auto& pos : originalEnemyPositions) {
390 enemyPositions.push_back(pos);
391 }
392
393 std::random_device rd;
394 std::mt19937 gen(rd());
395
396 std::shuffle(allValidPositions.begin(), allValidPositions.end(), gen);
397
398 for (const auto& pos : allValidPositions) {
399 if (std::find(enemyPositions.begin(), enemyPositions.end(), pos) != enemyPositions.end()) {
400 continue;
401 }
402
403 if (oldPlayer) {
404 Point playerPos = oldPlayer->getPosition();
405 int distance = abs(pos.x - playerPos.x) + abs(pos.y - playerPos.y);
406 if (distance < 5) {
407 continue;
408 }
409 }
410
411 enemyPositions.push_back(pos);
412
413 if (enemyPositions.size() >= static_cast<size_t>(adjustedEnemyCount)) {
414 break;
415 }
416 }
417
418 if (enemyPositions.size() < static_cast<size_t>(adjustedEnemyCount)) {
419 for (const auto& pos : allValidPositions) {
420 if (std::find(enemyPositions.begin(), enemyPositions.end(), pos) == enemyPositions.end()) {
421 enemyPositions.push_back(pos);
422 if (enemyPositions.size() >= static_cast<size_t>(adjustedEnemyCount)) {
423 break;
424 }
425 }
426 }
427 }
428
429 if (enemyPositions.size() > static_cast<size_t>(adjustedEnemyCount)) {
430 enemyPositions.resize(adjustedEnemyCount);
431 }
432
433 // РАСПРЕДЕЛЕНИЕ ТИПОВ ТАНКОВ В ЗАВИСИМОСТИ ОТ УРОВНЯ
434 std::vector<EnemyTankType> tankTypes;
435
436 for (const auto& pos : enemyPositions) {
437 EnemyTankType tankType = getRandomTankType(level, gen);
438 tankTypes.push_back(tankType);
439 }
440
441 // СОЗДАЕМ ВРАЖЕСКИЕ ТАНКИ С РАЗНЫМИ ТИПАМИ
442 for (size_t i = 0; i < enemyPositions.size(); i++) {
443 AIBehavior behavior = getAIBehaviorForDifficulty(enemyDifficulty, gen);
444 EnemyTankType tankType = tankTypes[i];
445
446 auto enemy = std::make_unique<EnemyTank>(enemyPositions[i], behavior, enemyDifficulty, tankType);
447 objects.push_back(std::move(enemy));
448 enemyCount++;
449 }
450
451 for (int y = 0; y < map.height; y++) {
452 for (int x = 0; x < map.width; x++) {
453 char cell = map.layout[y][x];
454 Point pos(x, y);
455
456 if (oldPlayer && pos.x == oldPlayer->getPosition().x && pos.y == oldPlayer->getPosition().y) {
457 continue;
458 }
459
460 if (std::find(enemyPositions.begin(), enemyPositions.end(), pos) != enemyPositions.end()) {
461 continue;
462 }
463
464 switch (cell) {
465 case '#': // Кирпичная стена
466 objects.emplace_back(new Obstacle(pos, ObstacleType::BRICK));
467 obstacleCount++;
468 break;
469 case 'X': // Стальная стена
470 objects.emplace_back(new Obstacle(pos, ObstacleType::STEEL));
471 obstacleCount++;
472 break;
473 case '~': // Вода
474 objects.emplace_back(new Obstacle(pos, ObstacleType::WATER));
475 obstacleCount++;
476 break;
477 case '*': // Лес
478 objects.emplace_back(new Obstacle(pos, ObstacleType::FOREST));
479 obstacleCount++;
480 break;
481 case '^':
482 case 'v':
483 case '<':
484 case '>':
485 if (oldPlayer && (y != oldPlayer->getPosition().y || x != oldPlayer->getPosition().x)) {
486 oldPlayer->setPosition(pos);
487 Direction dir = Direction::UP;
488 if (cell == 'v') dir = Direction::DOWN;
489 else if (cell == '<') dir = Direction::LEFT;
490 else if (cell == '>') dir = Direction::RIGHT;
491 oldPlayer->rotate(dir);
492 }
493 break;
494 default:
495 break;
496 }
497 }
498 }
499}

501EnemyTankType MapManager::getRandomTankType(int level, std::mt19937& gen) const {
502 std::uniform_int_distribution<> typeDist(0, 100);
503 int typeRoll = typeDist(gen);
504
505 if (level >= 2) {
506 if (typeRoll < 50) { // 50% - обычные танки
507 return EnemyTankType::BASIC;
508 } else if (typeRoll < 70) { // 20% - быстрые танки
509 return EnemyTankType::FAST;
510 } else if (typeRoll < 85) { // 15% - танки с уроном
511 return EnemyTankType::DAMAGE;
512 } else { // 15% - бронированные танки
513 return EnemyTankType::ARMORED;
514 }
515 } else {
516 // На первом уровне только обычные танки
517 return EnemyTankType::BASIC;
518 }
519
520 // На высоких уровнях увеличиваем шанс появления специальных танков
521 if (level >= 4) {
522 typeRoll = typeDist(gen); // Перебрасываем кубик
523 if (typeRoll < 30) { // 30% - обычные
524 return EnemyTankType::BASIC;
525 } else if (typeRoll < 55) { // 25% - быстрые
526 return EnemyTankType::FAST;
527 } else if (typeRoll < 75) { // 20% - с уроном
528 return EnemyTankType::DAMAGE;
529 } else { // 25% - бронированные
530 return EnemyTankType::ARMORED;
531 }
532 }
533
534 return EnemyTankType::BASIC;
535}

538 // Проверяем границы

540 pos.y <= 0 || pos.y >= map.height - 1 ||
541 pos.x == map.width / 2 || pos.y == map.height - 3) {
542

544 }
545

547 Point playerPos = player->getPosition();
548 int distance = abs(pos.x - playerPos.x) + abs(pos.y - playerPos.y);

557 case 'X':
558 case '~':

560 return false;
561 case '*':

563 case '^':
564 case 'v':

599AIBehavior MapManager::getAIBehaviorForDifficulty(int difficulty, std::mt19937& gen) const {
600 std::uniform_real_distribution<> dist(0.0, 1.0);
601 double random = dist(gen);
602
603 switch (difficulty) {
604 case 1:

606 if (random < 0.9) return AIBehavior::DEFENSIVE;
607 return AIBehavior::AGGRESSIVE;
608

611 if (random < 0.7) return AIBehavior::DEFENSIVE;
612 return AIBehavior::AGGRESSIVE;
613
614 case 3:
615 if (random < 0.2) return AIBehavior::RANDOM;
616 if (random < 0.5) return AIBehavior::DEFENSIVE;
617 return AIBehavior::AGGRESSIVE;
618
619 default:
620 return AIBehavior::RANDOM;
621 }
