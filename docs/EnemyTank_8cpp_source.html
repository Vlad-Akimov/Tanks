116Direction EnemyTank::getDirectionToPlayer() const {
117 if (playerLastPosition.x == -1) return direction;
118
119 int dx = playerLastPosition.x - position.x;
120 int dy = playerLastPosition.y - position.y;
121
122 if (abs(dx) > abs(dy)) {
123 return (dx > 0) ? Direction::RIGHT : Direction::LEFT;
124 } else {
125 return (dy > 0) ? Direction::DOWN : Direction::UP;
126 }

146 // Вражеские танки имеют разную точность в зависимости от сложности
147 if (difficulty == 1 && rand() % 3 == 0) { // 33% шанс промаха
148 // Стреляем в случайном направлении
149 Direction randomDir = static_cast<Direction>(rand() % 4);
150 rotate(randomDir);
151 } else if (difficulty == 2 && rand() % 5 == 0) { // 20% шанс промаха
152 // Небольшое отклонение
153 Direction nearDir = getDirectionToPlayer();
154 if (rand() % 2 == 0) {
155 // Слегка отклоняемся
156 switch (nearDir) {
157 case Direction::UP:
158 case Direction::DOWN:
159 rotate(rand() % 2 == 0 ? Direction::LEFT : Direction::RIGHT);
160 break;
161 case Direction::LEFT:
162 case Direction::RIGHT:
163 rotate(rand() % 2 == 0 ? Direction::UP : Direction::DOWN);
164 break;
165 }
166 }
167 }
168 // Высокая сложность - всегда стреляет точно к игроку
169 else if (difficulty == 3) {
170 rotate(getDirectionToPlayer());
171 }
172
173 // Создаем снаряд с соответствующим уроном
174 Projectile* projectile = Tank::fire();
175 if (projectile) {
176 projectile->setDamage(projectileDamage);
177 }

331void EnemyTank::attemptShot() {
332 if (!canFire()) return;
333
334 int shotChance = 0;
335
336 // Вероятность выстрела в зависимости от поведения и сложности
337 switch (behavior) {
338 case AIBehavior::AGGRESSIVE:
339 shotChance = 70 + (difficulty * 10); // 70-90%
340 break;
341 case AIBehavior::DEFENSIVE:
342 shotChance = 40 + (difficulty * 10); // 40-60%
343 break;
344 case AIBehavior::RANDOM:
345 shotChance = 30 + (difficulty * 10); // 30-50%
346 break;
347 }
348
349 if (hasClearShot() && (rand() % 100 < shotChance)) {
350 rotate(getDirectionToPlayer());
351 fire();
352 }
