259

265 }

270

293 // Используем графическую карту для улучшенного отображения

300 std::cout << symbol;

302
303 resetColor();
304 }
305
306 setColor(PlatformUtils::Color::BLACK);

322
323 // Отображаем статус бонусов игрока
324 PlayerTank* player = world.getPlayer();
325 if (player) {
326 std::string bonusLine = "Bonuses: ";
327 if (player->getHasShield()) {
328 bonusLine += "[★ Shield] ";
329 }
330 if (player->getDoubleFire()) {
331 bonusLine += "[¶ Double Fire] ";
332 }
333 if (player->getBonusDuration() > 0) {
334 bonusLine += "(turns left: " + std::to_string(player->getBonusDuration()) + ")";
335 }
336
337 // Центрируем строку бонусов
338 int bonusOffset = calculateHorizontalOffset(bonusLine.length());
339 std::cout << std::string(bonusOffset, ' ') << bonusLine << "\n";
340 }
341
342 drawSymbolLegend();
343
344 return true;
345}
