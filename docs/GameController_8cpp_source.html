60void GameController::showMapSelection() {
61 bool inMapSelection = true;
62
63 if (mapManager.getMapCount() == 0) {
64 mapManager.loadMaps();
65 }
66
67 if (mapManager.getMapCount() == 0) {
68 std::cout << "Maps not found! Using random generation." << std::endl;
69 useCustomMap = false;
70 model.loadLevel(1);
71 model.setState(GameState::PLAYING);
72 return;
73 }
74
75 while (inMapSelection && running) {
76 handleTerminalResize();
77 if (!running) return;
78
79 const MapInfo& currentMap = mapManager.getMap(currentMapIndex);
80
81 view.clearScreen();
82 if (!view.drawMapSelection(currentMap, currentMapIndex, mapManager.getMapCount())) {
83 continue;
84 }
85 std::cout.flush();
86
87 std::string input = PlatformUtils::readUTF8Char();
88
89 if (input == "a" || input == "A" || input == "ф" || input == "Ф") {
90 // Предыдущая карта
91 currentMapIndex--;
92 if (currentMapIndex < 0) {
93 currentMapIndex = mapManager.getMapCount() - 1;
94 }
95 }
96 else if (input == "d" || input == "D" || input == "в" || input == "В") {
97 // Следующая карта
98 currentMapIndex++;
99 if (currentMapIndex >= mapManager.getMapCount()) {
100 currentMapIndex = 0;
101 }
102 }
103 else if (input == "f" || input == "F" || input == "а" || input == "А") {
104 // Случайная генерация
105 useCustomMap = false;
106 model.loadLevel(1);
107 model.setState(GameState::PLAYING);
108 inMapSelection = false;
109 }
110 else if (input == "\n" || input == " ") {
111 // Подтверждение выбора карты
112 useCustomMap = true;
113 loadSelectedMap();
114 inMapSelection = false;
115 }
116 else if (input == "m" || input == "M" || input == "ь" || input == "Ь") {
117 // В меню
118 inMapSelection = false;
119 showMenu();
120 return;
121 }
122 else if (input == "q" || input == "Q" || input == "й" || input == "Й") {
123 // Выход
124 running = false;
125 inMapSelection = false;
126 }

216void GameController::processGameTurn() {
217 handleTerminalResize();
218 if (!running) return;
219
220 view.clearScreen();
221 int score = 0;
222
223 switch (model.getState()) {
224 case GameState::PLAYING:
225 if (!view.render(model)) {
226 // Если рендер не удался из-за размера терминала, выходим из цикла
227 return;
228 }
229 break;
230
231 case GameState::LEVEL_COMPLETE:
232 showLevelCompleteScreen();
233 return;
234
235 case GameState::PAUSED:
236 view.drawPauseScreen();
237 break;
238
239 case GameState::GAME_OVER:
240 // Сохраняем рекорд только один раз при Game Over
241 if (model.getPlayer() && !scoreSaved) {
242 score = model.getPlayer()->getScore();
243 if (score > 0) {
244 scoreManager.addScore(score);
245 scoreSaved = true; // Помечаем как сохраненный
246 }
247 }
248 view.drawGameOver(model.getPlayer() ? model.getPlayer()->getScore() : 0);
249 break;
250
251 default:
252 break;
253 }
254
255 std::cout.flush();
256
257 if (model.getState() == GameState::LEVEL_COMPLETE) {
258 return;
259 }
260
261 Command cmd = inputHandler.waitForCommand();
262 processCommand(cmd);
263
264 if (model.getState() == GameState::PLAYING &&
265 (cmd == Command::MOVE_UP || cmd == Command::MOVE_DOWN ||
266 cmd == Command::MOVE_LEFT || cmd == Command::MOVE_RIGHT ||
267 cmd == Command::FIRE)) {

277void GameController::processCommand(Command cmd) {
278 switch (cmd) {
279 case Command::MOVE_UP:
280 if (model.getState() == GameState::PLAYING) {
281 model.playerMove(Direction::UP);
282 }
283 break;
284
285 case Command::MOVE_DOWN:
286 if (model.getState() == GameState::PLAYING) {
287 model.playerMove(Direction::DOWN);
288 }
289 break;
290
291 case Command::MOVE_LEFT:
292 if (model.getState() == GameState::PLAYING) {
293 model.playerMove(Direction::LEFT);
294 }
295 break;
296
297 case Command::MOVE_RIGHT:
298 if (model.getState() == GameState::PLAYING) {
299 model.playerMove(Direction::RIGHT);
300 }
301 break;
302
303 case Command::FIRE:
304 if (model.getState() == GameState::PLAYING) {
305 model.playerFire();
306 }
307 break;
308
309 case Command::PAUSE:
310 pauseGame();
311 break;
312
313 case Command::MENU:
314 if (model.getState() == GameState::PLAYING) {
315 // Сохраняем рекорд при выходе в меню из игры
316 if (model.getPlayer() && model.getPlayer()->getScore() > 0 && !scoreSaved) {
317 scoreManager.addScore(model.getPlayer()->getScore());
318 scoreSaved = true;
319 }
320 model.setState(GameState::GAME_OVER);
321 model.setCurrentLevel(1);
322 showMenu();
323 } else if (model.getState() == GameState::GAME_OVER) {
324 model.setCurrentLevel(1);
325 showMenu();
326 }
327 if (model.getPlayer()) {
328 model.getPlayer()->setLives(3);
329 }
330 break;
331
332 case Command::CONFIRM:
333 if (model.getState() == GameState::MENU) {
334 // Сбрасываем флаг при начале новой игры
335 scoreSaved = false;
336 useCustomMap = false;
337 model.setCurrentLevel(1);
338 if (model.getPlayer()) {
339 model.getPlayer()->setScore(0);
340 model.getPlayer()->setLives(3);
341 }
342 showMapSelection();
343 } else if (model.getState() == GameState::PAUSED) {
344 model.setState(GameState::PLAYING);
345 } else if (model.getState() == GameState::GAME_OVER) {
346 // Сохраняем рекорд при подтверждении на экране Game Over
347 if (model.getPlayer() && model.getPlayer()->getScore() > 0 && !scoreSaved) {
348 scoreManager.addScore(model.getPlayer()->getScore());
349 scoreSaved = true;
350 }
351 // Сбрасываем флаг для новой игры
352 scoreSaved = false;
353 useCustomMap = false;
354 model.setCurrentLevel(1);
355 if (model.getPlayer()) {
356 model.getPlayer()->setScore(0);
357 model.getPlayer()->setLives(3);
358 }
359 showMapSelection();
360 model.setState(GameState::PLAYING);
361 } else if (model.getState() == GameState::LEVEL_COMPLETE) {
362 // Обработка подтверждения на экране завершения уровня
363 }
364 break;
365
366 case Command::BACK:
367 if (model.getState() == GameState::SETTINGS) {
368 model.setState(GameState::MENU);
369 } else if (model.getState() == GameState::PAUSED) {
370 model.setState(GameState::PLAYING);
371 }
372 break;
373
374 case Command::EXIT:
375 if (model.getPlayer() && model.getPlayer()->getScore() > 0 && !scoreSaved) {
376 scoreManager.addScore(model.getPlayer()->getScore());
377 scoreSaved = true;
378 }
379 running = false;
380 break;
381
382 case Command::NONE:
383 break;
384 }
